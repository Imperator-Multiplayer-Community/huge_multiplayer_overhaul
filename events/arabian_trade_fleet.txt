                    NOT = { trade_goods = gems }

                    NOT = { trade_goods = spices }
                    NOT = { trade_goods = silk }
                } 
                set_trade_goods = silk
                }
                create_country_treasure = {
                    key = "heshibis_jade_disc"
                    icon = "treasure_chest"
                
                    modifier = {
                        local_research_points_modifier = 0.1
