				trigger = {
					hof_disapproves_age = yes
				}
				desc = "hof_disapproves_age_desc"

					hof_approves = yes

					is_minor_character = no

					}
				}
			}

				save_scope_as = negotiator
			}
		}
		set_variable = {
			name = suitor_age 
			value = scope:suitor.age
		}

		}

			trigger_else = {

			marry_character = scope:love_target
		}

			add_loyalty = loyalty_marriage_arranged_1
		}

			limit = {
				exists = scope:negotiator
			}
			scope:negotiator = {
				add_loyalty = loyalty_marriage_arranged_2
			}
		}
	}

		trigger = {
			trigger_if = {
				limit = {
					exists = scope:negotiator

					self_disapproves_loyalty = yes

			limit = {

				}

			scope:love_target = {
				add_gold = {
					value = root.has_monthly_income
					multiply = 3
				}
			}
		}

			value = has_monthly_income
			multiply = -3
		}

			if = {	
				limit = {
					scope:love_target = {
						is_married = yes
					}
				}
				scope:love_target = {
					divorce_character = spouse
				}
			}
			if = {	
				limit = {
					scope:suitor = {
						is_married = yes
					}
				}
				scope:suitor = {
					divorce_character = spouse

			marry_character = scope:love_target

			add_loyalty = loyalty_marriage_arranged_1
		}
		if = {
			limit = {
				exists = scope:negotiator
			}
			scope:negotiator = {
				add_loyalty = loyalty_marriage_arranged_2
			}
		}
	}

		name = "royal_marriage.6.c"
		trigger = {
			trigger_if = {
				limit = {
					exists = scope:negotiator
				}
				hof_disapproves_loyalty = yes
				NOT = {
					hof_disapproves_age = yes
				}
			}
			trigger_else = {
				self_disapproves_loyalty = yes
				NOT = {
					self_disapproves_age = yes
				}
			}

			limit = {

					value = root.has_monthly_income

				}
			}
		}
		add_treasury = {
			value = has_monthly_income
			multiply = -6
		}
		hidden_effect = {
			if = {	
				limit = {
					scope:love_target = {
						is_married = yes
					}
				}
				scope:love_target = {
					divorce_character = spouse
				}
			}
			if = {	
				limit = {
					scope:suitor = {
						is_married = yes
					}
				}
				scope:suitor = {
					divorce_character = spouse
				}
			}
		}
		scope:suitor = {
			marry_character = scope:love_target
		}
		scope:love_target = {
			add_loyalty = loyalty_marriage_arranged_1
		}
		if = {
			limit = {
				exists = scope:negotiator
			}
			scope:negotiator = {
				add_loyalty = loyalty_marriage_arranged_2
			}
		}
	}

		name = "royal_marriage.6.d"
		custom_tooltip = unable_to_propose_to_for_5_years
