## adds chariots discipline distinctions
ml_chariots_distinctions.5 = {
	type = country_event
	hidden = yes

	trigger = {
		scope:actor = {
			any_sub_unit = {
            	count >= 20
            	AND = {
	            	sub_unit_type = chariots
            	}
            }
			has_legion_trigger = yes
		}
	}

	immediate = {
		scope:actor = {
			if = {
				limit = {
					NOT = {
						has_variable = chariots_victories
					}
				}

				set_variable = {
					name = chariots_victories
					value = 1
				}
			}
			else = {
				change_variable = {
					name = chariots_victories
					add = 1
				}
			}

			if = {
				limit = {
					var:chariots_victories >= 4
					## var:chariots_victories >= 1 ## testing
				}

				if = {
					limit = {
						AND = { 
							var:chariots_victories >= 4
							var:chariots_victories < 10
							## var:chariots_victories >= 1 ## testing
							NOT = {
								scope:actor.legion = {
									OR = {
										has_distinction = distinction_chariots_discipline_1
										has_distinction = distinction_chariots_discipline_2
									}
								}
							}
						}
						
					}

					scope:actor.legion = {
						add_distinction = distinction_chariots_discipline_1
					}

					ROOT = {
						trigger_event = {
							id = ml_expeditionary_distinctions.1
						}
					}
				}
				else_if = {
					limit = {
						AND = { 
							var:chariots_victories >= 10
							var:chariots_victories < 20
							## var:chariots_victories >= 1 ## testing
							NOT = {
								scope:actor.legion = {
									OR = {
										has_distinction = distinction_chariots_discipline_2
										has_distinction = distinction_chariots_discipline_3
									}
								}
							}
						}
						
					}

					scope:actor.legion = {
						remove_distinction = distinction_chariots_discipline_1
						add_distinction = distinction_chariots_discipline_2
					}

					ROOT = {
						trigger_event =  {
							id = ml_cohort_master_distinctions.1
						}
					}
				}
				else_if = {
					limit = {
						AND = {
							var:chariots_victories >= 20
							## var:chariots_victories >= 2 ## testing
							scope:actor.legion = {
								has_distinction = distinction_chariots_discipline_2
								NOT = {
									has_distinction = distinction_chariots_discipline_3
								}
							}
						}
					}
					
					scope:actor.legion = {
						remove_distinction = distinction_chariots_discipline_2
						add_distinction = distinction_chariots_discipline_3
					}

					ROOT = {
						trigger_event =  {
							id = ml_cohort_master_distinctions.1
						}
					}
				}
			}
			
		}
	}
}

## removes chariots discipline victory count
ml_chariots_distinctions.6 = {
	type = country_event
	hidden = yes

	trigger = {
		scope:actor = {
			any_sub_unit = {
            	count >= 1
            	AND = {
	            	sub_unit_type = chariots
            	}
            }
			has_legion_trigger = yes
		}
	}

	immediate = {
		scope:actor = {
			if = {
				limit = {
					has_variable = chariots_victories
				}

				if = {
					limit = {
						var:chariots_victories >= 20
					}

					set_variable = {
						name = chariots_victories
						value = 20
					}
				}
				else_if = {
					limit = {
						var:chariots_victories >= 10
					}

					set_variable = {
						name = chariots_victories
						value = 10
					}
				}
				else_if = {
					limit = {
						var:chariots_victories >= 4
					}

					set_variable = {
						name = chariots_victories
						value = 4
					}
				}
				else = {
					remove_variable = chariots_victories
				}
			}
		}
	}
