formable_heritage.1 = {
	type = country_event	# EVENT TYPE - Can be Country/Province/Character, decides ROOT Scope
	title = formable_heritage.1.t	# LOCALIZATION KEY - Title of event
	desc = formable_heritage.1.desc	# LOCALIZATION KEY - Description of event
	picture = aqueducts	# EVENT PICTURE
	left_portrait = current_ruler	# LEFT PORTRAIT - Ruler shown at the left hand side in event
	trigger = {
		# TRIGGER FIELD - Checks if the event will fire at all
		tag = LNK
	}
	immediate = {
		save_scope_as = lankan_country
	}
	option = {
		# EFFECT FIELD - FIRST OPTION
		name = formable_heritage.1.a		# LOCALIZATION KEY - Option A text
		hidden_effect = {
			set_country_heritage = lankan_heritage			# EFFECT
		}
		show_as_tooltip = {
			add_country_modifier = {
				name = lankan_modifier
				duration = -1
			}
		}
	}
	option = {
		# EFFECT FIELD - SECOND OPTION
		name = formable_heritage.1.b		# LOCALIZATION KEY - Option B text
		add_political_influence = 75
	}
	option = {
		# EFFECT FIELD - THIRD OPTION
		name = formable_heritage.1.c		# LOCALIZATION KEY - Option B text
		hidden_effect = {
			set_country_heritage = clan_heritage
		}
		show_as_tooltip = {
			add_country_modifier = {
				name = clan_modifier
				duration = -1
			}
		}
	}
}
