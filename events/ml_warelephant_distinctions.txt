## adds warelephant discipline distinctions
ml_warelephant_distinctions.5 = {
	type = country_event
	hidden = yes

	trigger = {
		scope:actor = {
			any_sub_unit = {
            	count >= 20
            	AND = {
	            	sub_unit_type = warelephant
            	}
            }
			has_legion_trigger = yes
		}
	}

	immediate = {
		scope:actor = {
			if = {
				limit = {
					NOT = {
						has_variable = warelephant_victories
					}
				}

				set_variable = {
					name = warelephant_victories
					value = 1
				}
			}
			else = {
				change_variable = {
					name = warelephant_victories
					add = 1
				}
			}

			if = {
				limit = {
					var:warelephant_victories >= 10
					## var:warelephant_victories >= 1 ## testing
				}

				if = {
					limit = {
						AND = { 
							var:warelephant_victories >= 10
							## var:warelephant_victories >= 1 ## testing
							NOT = {
								scope:actor.legion = {
									OR = {
										has_distinction = distinction_warelephant_discipline_1
										has_distinction = distinction_warelephant_discipline_2
									}
								}
							}
						}
						
					}
					
					trigger_event = {
						id = ml_cohort_master_distinctions.1
					}

					scope:actor.legion = {
						add_distinction = distinction_warelephant_discipline_1
					}
				}
				else_if = {
					limit = {
						AND = {
							var:warelephant_victories >= 20
							## var:warelephant_victories >= 2 ## testing
							scope:actor.legion = {
								has_distinction = distinction_warelephant_discipline_1
								NOT = {
									has_distinction = distinction_warelephant_discipline_2
								}
							}
						}
					}

					trigger_event =  {
						id= ml_cohort_master_distinctions.1
					}
					
					scope:actor.legion = {
						remove_distinction = distinction_warelephant_discipline_1
						add_distinction = distinction_warelephant_discipline_2 
					}
				}
			} 
			
		}
	}
}
