namespace = ml_expeditionary_distinctions

# adds expeditionary count
ml_expeditionary_distinctions.1 = {
	type = country_event
	hidden = yes

	trigger = {
		scope:actor = {
			has_legion_trigger = yes
		}
	}

	immediate = {
		scope:actor = {
			if = {
				limit = {
					has_variable = expeditionary
				}
				change_variable = {
					name = expeditionary
					add = 1
				}
			}
			else = {
				set_variable = {
					name = expeditionary
					value = 1
				}
			}
			
			ROOT = {
				trigger_event = {
					id = ml_expeditionary_distinctions.3
				}
			}
		}
	}
}

# removes expeditionary count
ml_expeditionary_distinctions.2 = {
	type = country_event
	hidden = yes

	trigger = {
		scope:actor = {
			has_legion_trigger = yes
		}
	}

	immediate = {
		scope:actor = {
			if = {
				limit = {
					has_variable = expeditionary
				}

				if = {
					limit = {
						var:expeditionary >= 1
					}

					change_variable = {
						name = expeditionary
						subtract = 1
					}
				}
			}
		}
	}
}

## adds any expeditionary distinctions
ml_expeditionary_distinctions.3 = {
	type = country_event
	hidden = yes

	trigger = {
    	scope:actor = {
            has_legion_trigger = yes
        	has_variable = expeditionary
        	NOT = {
        		legion = {
					has_distinction = distinction_expeditionary_01
        		}
			}
        }
	}

	immediate = {
		scope:actor = {
			if = {
				limit = {
					var:expeditionary >= 20
					# testing
					# var:expeditionary >= 2
				}

				scope:actor.legion = {
					add_distinction = distinction_expeditionary_01
				}
			}
		}
	}
}
