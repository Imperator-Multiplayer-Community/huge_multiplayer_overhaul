hunting.1 = { #You send the invite to a ruler. Does he accept to correspond? Received by target employer
	type = country_event
	title = "hunting.1.t"
	desc = "hunting.1.desc"
	left_portrait = current_ruler
	right_portrait = scope:actor.current_ruler
	picture = steppe_cavalry
	
	option = { #Option for recipient to accept
		name = "hunting.1.a"
		scope:target = {
			set_variable = {
				name = hunting_friend
				value = scope:actor.current_ruler
				days = 365
			}
			save_scope_as = hunting_friend
		}
		scope:actor.current_ruler = {
			set_variable = {
				name = hunter
				value = scope:target
				days = 365
			}
			save_scope_as = hunter
		}
		custom_tooltip = friend_refusal
		hidden_effect = {
			if = {
				limit = {
					scope:actor = {
						has_land = yes
					}
				}
				scope:actor = {
					trigger_event = {
						id = hunting.3
						days = 1
					}
				}
			}
		}
		# ai always should accept
		ai_chance = {
			factor = 100
		}
	}
	
	option = { #Option for recipient to decline
		name = "hunting.1.b"
		custom_tooltip = friend_refusal
		hidden_effect = {
			if = {
				limit = {
					scope:actor = {
						has_land = yes
					}
				}
				scope:actor = {
					trigger_event = {
						id = hunting.2
						days = 1
					}
				}
			}
		}
		ai_chance = {
			factor = 0
		}
	}
}
