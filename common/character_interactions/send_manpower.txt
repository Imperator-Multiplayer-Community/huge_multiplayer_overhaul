send_men_1k = {
	on_other_nation = yes
	on_own_nation = no
	message = yes

	potential_trigger = {
		scope:target = {
			NOT = { employer = scope:actor }
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }	
		}
		scope:actor = {
			is_ai = no
		}
		scope:actor.current_ruler ={
			trigger_if = {
				limit = {
					has_variable = manpoweropen
				}
				var:manpoweropen = 0
			}
			trigger_else = {
				always = no
			}
		}

	}

	allowed_trigger = {
		scope:actor = {
			can_pay_price = onekman
			trigger_if = {
				limit = {
					has_variable = manpowercooldown
				}
				custom_tooltip = {
					text = manpowercd
					NOT = {var:manpowercooldown = 1}
				}
			}
		}
		scope:target.employer = {
			in_diplomatic_range = scope:actor
			opinion = {
        		target = scope:actor
        		value >= 100
        	}
		}
		scope:target = {
			is_friend = scope:actor.current_ruler
		}
		
	}

	effect = {
		scope:actor = {
			set_variable = {
				name = manpowercooldown
				value = 1
				days = 90
			}
			pay_price = onekman
		}
		scope:target.employer = {
			add_manpower = 2
		}
		scope:actor.current_ruler = {
			set_variable = {
				name = manpoweropen
				value = 1
			}
		}
	}

}

send_men_open = {
	on_other_nation = yes
	on_own_nation = no
	message = no

	potential_trigger = {
		scope:actor = {
			is_ai = no
		}
		scope:actor.current_ruler ={
			OR = {
				NOT = {
					has_variable = manpoweropen
				}
				var:manpoweropen = 1
			}
		}
		scope:target = {
			NOT = { employer = scope:actor }
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }	
		}

	}
	allowed_trigger = {
		scope:actor.current_ruler ={
			OR = {
				NOR = {
					has_variable = manpoweropen
					has_variable = manpowercooldown
				}
				var:manpoweropen = 1
			}
		}
		scope:target.employer = {
			in_diplomatic_range = scope:actor
		}
		
	}

	effect = {
		scope:actor.current_ruler = {
			set_variable = {
				name = manpoweropen
				value = 0
			}
		}
	}
}
