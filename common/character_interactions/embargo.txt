embargo = {
	on_other_nation = yes
	on_own_nation = no
	message = no

	potential_trigger = {
		hidden:scope:target = {
			is_mercenary = no
			employer = {
				NOR = {
					tag = PIR
					tag = REB
					tag = BAR
				}
			}
			is_alive = yes
			is_ruler = yes
			NOT = {
				this = scope:actor.current_ruler
			}

			trigger_if = {
				limit = {
					scope:actor = {
						has_variable_list = embargo_list
					}
				}

				scope:actor = {
					NOT = {
						is_target_in_variable_list = {
							name = embargo_list
							target = scope:target.employer
						}
					}
				}
			}
		}
	}

	allowed_trigger = {
		scope:target = {
			is_ruler = yes
		}
		scope:actor = {
			in_diplomatic_range = scope:target.employer
		}
		scope:target.employer = {
			opinion = {
				target = scope:actor
				value < 0
			}
		}
	}

	effect = {
		scope:actor = {
			add_to_variable_list = {
				name = embargo_list
				target = scope:target.employer
			}
		}

		scope:target.employer = {
			add_to_variable_list = {
				name = embargoed_list
				target = scope:actor
			}
			
			trigger_event = {
				id = embargo.1
			}
		}
	}
}

lift_embargo = {
	on_other_nation = yes
	on_own_nation = no
	message = no

	potential_trigger = {
		hidden:scope:target = {
			is_mercenary = no
			employer = {
				NOR = {
					tag = PIR
					tag = REB
					tag = BAR
				}
			}
			is_alive = yes
			is_ruler = yes
			NOT = {
				this = scope:actor.current_ruler
			}
		}

		AND = {
			scope:actor = {
				has_variable_list = embargo_list
			}
			scope:actor = {
				is_target_in_variable_list = {
					name = embargo_list
					target = scope:target.employer
				}
			}
		}
	}

	allowed_trigger = {
		scope:target = {
			is_ruler = yes
		}
	}

	effect = {
		scope:actor = {
			remove_list_variable = {
				name = embargo_list
				target = scope:target.employer
			}
		}

		scope:target.employer = {
			remove_list_variable = {
				name = embargoed_list
				target = scope:actor
			}
			
			trigger_event = {
				id = embargo.2
			}
		}
	}
}
