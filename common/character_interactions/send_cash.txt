send_cash_500 = {
	on_other_nation = yes
	on_own_nation = no
	message = yes

	potential_trigger = {
		scope:target = {
			NOT = { employer = scope:actor }
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }	
		}
		scope:actor = {
			is_ai = no
		}
		scope:actor.current_ruler ={
			trigger_if = {
				limit = {
					has_variable = cashopen
				}
				var:cashopen = 0
			}
			trigger_else = {
				always = no
			}
		}

	}

	allowed_trigger = {
		scope:actor = {
			can_pay_price = cash500
			trigger_if = {
				limit = {
					has_variable = cashcooldown
				}
				custom_tooltip = {
					text = cashcd
					NOT = {var:cashcooldown = 1}
				}
			}
		}
		scope:target.employer = {
			in_diplomatic_range = scope:actor
			opinion = {
        		target = scope:actor
        		value >= 100
        	}
		}
		scope:target = {
			is_friend = scope:actor.current_ruler
		}
		
	}

	effect = {
		scope:actor = {
			set_variable = {
				name = cashcooldown
				value = 1
				days = 90
			}
			pay_price = cash500
		}
		scope:target.employer = {
			add_treasury = 500
		}
		scope:actor.current_ruler = {
			set_variable = {
				name = cashopen
				value = 1
			}
		}
	}

}

send_cash_open = {
	on_other_nation = yes
	on_own_nation = no
	message = no

	potential_trigger = {
		scope:actor = {
			is_ai = no
		}
		scope:actor.current_ruler ={
			OR = {
				NOT = {
					has_variable = cashopen
				}
				var:cashopen = 1
			}
		}
		scope:target = {
			NOT = { employer = scope:actor }
			NOT = { employer = { tag = REB } }
			NOT = { employer = { tag = BAR } }	
		}

	}
	allowed_trigger = {
		scope:actor.current_ruler ={
			OR = {
				NOR = {
					has_variable = cashopen
					has_variable = cashcooldown
				}
				var:cashopen = 1
			}
		}
		scope:target.employer = {
			in_diplomatic_range = scope:actor
		}
		
	}

	effect = {
		scope:actor.current_ruler = {
			set_variable = {
				name = cashopen
				value = 0
			}
		}
	}
}
