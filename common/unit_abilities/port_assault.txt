port_assault = {
	duration = 61
	enable = yes
	toggle = yes
	navy_only = yes

	soundeffect = "event:/SFX/UI/Unit/sfx_ui_unit_port_assualt"
	map = no

	allow = {
		is_navy = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		custom_tooltip = {
			text = port_assault_explain_tt
			any_sub_unit = {
				sub_unit_type = mega_galley
				subunit_strength_percentage >= 0.95
			}
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
			unit_location = {
				any_neighbor_province = {
					is_sea = no
					has_owner = yes
					is_port = yes
					fort_level >= 1
					NOT = { controller = root.unit_owner }
					controller = { war_with = root.unit_owner }
					NOT = { has_enemy_army = root.unit_owner }
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = port_assault_no_valid_port
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						fort_level >= 1
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						NOT = { has_enemy_army = root.unit_owner }
					}
				}
			}
		}
	}
	
	start_effect = {
		save_scope_as = naval_assaulter_unit

		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						can_assault_port = yes
					}
				}
			}
 
			custom_tooltip = port_assault_explainer 
			show_as_tooltip = { #All fake for the tooltips sake
				random_sub_unit = {
					limit = {
						sub_unit_type = mega_galley
						subunit_strength_percentage >= 0.95
					}
					add_subunit_strength = -0.70
				}
				
				add_unit_modifier = {
					name = naval_action
					duration = 60
				}
			}

			hidden_effect = {
				set_variable = is_navy_assaulting

				random_sub_unit = {
					limit = {
						sub_unit_type = mega_galley
						subunit_strength_percentage >= 0.95
					}
					add_subunit_strength = -0.70
				}

				add_unit_modifier = {
					name = naval_action
					duration = 60
				}

				unit_location = {
					save_scope_as = naval_assaulter_location
				}
				
				unit_owner = {
					trigger_event = {
						id = hmo_unit_abilties.4
						days = 60
					}
				}
			}
		}
		else = {
			custom_tooltip = port_assault_explainer
		}
	}

	finished_when = {
		OR = {
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			is_moving = yes
			has_commander = no
			strength_percentage < 0.3
			morale_percentage < 0.3
			in_combat = yes
		}
	}

	finish_effect = {
		navy_port_assault_finish_effect = yes
	}

	idle_entity_state = combat
}
