fishing = {
	duration = -1
	enable = yes
	toggle = yes
	navy_only = yes

	map = no

	soundeffect ="event:/SFX/UI/Unit/sfx_ui_unit_reorganization"

	allow = {
		is_navy = yes
		unit_size >= 10
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege = no
		has_commander = yes
		unit_owner = {
			NOT = {
				has_country_modifier = go_fish
				has_low_economic_policy = expense_navy
			}
		}

		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}

		trigger_if = {
			limit = {
				NOT = {
					unit_location = {
						any_neighbor_province = {
							trade_goods = fish
						}
					}
				}
			}

			custom_tooltip = {
				text = fishing_needs_fish
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						has_owner = yes
						is_sea = no
						NOT = {
							has_province_modifier = getting_fished
						}
						total_population <= 4
						trade_goods = fish
					}
				}
			}
			custom_tooltip = {
				text = fishing_not_enough_pops
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						has_owner = yes
						is_sea = no
						total_population > 4
						trade_goods = fish
						has_province_modifier = getting_fished
					}
				}
			}
			custom_tooltip = {
				text = fishing_cooldown_tooltip
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							fishing_province_trigger = yes
						}
					}
				}
			}
			custom_tooltip = {
				text = explain_fishing_tooltip
				always = no
			}
		}
		trigger_else = {
			unit_location = {
				any_neighbor_province = {
					fishing_province_trigger = yes
				}
			}
		}
	}


	start_effect = {
		custom_tooltip = fishing_explain_modifier_tooltip
		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						fishing_province_trigger = yes
					}
				}
			}

			hidden_effect = {
				set_variable = is_navy_fishing
			}

			navy_action_effect = yes

			unit_owner = {
				while = {
					count = root.num_of_ships_svalue
					add_country_modifier = {
						name = go_fish
						duration = -1
						mode = add
					}
				}
			}

			unit_location = {
				ordered_neighbor_province = {
					limit = {
						fishing_province_trigger = yes
						NOT = {	has_province_modifier = getting_fished }
					}
					order_by = total_population
					max = 1
					add_province_modifier = {
						name = getting_fished
						duration = -1
					}

					trigger_event = {
						id = hmo_unit_abilities.11
					}
				}
			}
		}
	}

	finished_when = {
		OR = {
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			is_moving = yes
			has_commander = no
			strength_percentage < 0.3
			morale_percentage < 0.3
			in_combat = yes
		}
	}

	finish_effect = {
		navy_fishing_finish_effect = yes
	}

	ai_will_do = {
		modifier = {
			trigger = {

				experience_percentage < 0.65

				unit_owner = {
					treasury > 50
					manpower > 20
				}
			}
			add = {
				value = 250
				divide = unit_size_rank
			}
		}
	}

	ai_will_revoke = {
		OR = {
			is_idle = no
			unit_threat > 0.2
			experience_percentage >= 0.95

			unit_owner = {
				treasury < 1
			}
			unit_owner = {
				months_to_war < 12
			}
		}
	}
}
