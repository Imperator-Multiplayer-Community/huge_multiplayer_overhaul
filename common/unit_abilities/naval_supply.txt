naval_supply = {

	duration = 90
	enable = yes
	toggle = no
	navy_only = yes
	cancel_on_combat_end = yes

	map = yes

	soundeffect ="event:/SFX/UI/Unit/sfx_ui_unit_reorganization"

	allow = {

		is_navy = yes
		unit_size >= 25
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege = no
		has_commander = yes
		custom_tooltip = {
			text = naval_supply_explain_tt
			any_sub_unit = {
				ship_category = heavy
				subunit_strength_percentage >= 0.95
			}
			# text = marines_ratio
			# num_of_marines >= num_of_ships_10
		}
		
		trigger_if = {
			limit = {
				unit_location = {
					NOT = {
							any_neighbor_province = {
								is_sea = no
								naval_supply_trigger = yes
							}
					}
						any_neighbor_province = {
							is_sea = no
							has_owner = yes
							is_port = yes
							has_siege = yes
							#owner = {
							#	NOT = { war = root.unit_owner }	
							#}
							NOT = {
								has_province_modifier = supplied_by_sea
							}
							owner = controller
						}	
				}
			}

		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							naval_supply_trigger = yes
						}
					}
				}
			}
			custom_tooltip = {
				text = explain_naval_supply_tooltip
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							naval_supply_trigger = yes
							NOT = {
								has_province_modifier = supplied_by_sea
							}
						}
					}
				}
			}
			custom_tooltip = {
				text = naval_supply_cooldown_tooltip
				always = no
			}
		}

		NOT = {
			OR = {
				unit_owner = { has_country_modifier = naval_supplier }  #New addition, lets see if it works
				unit_owner = { has_low_economic_policy = expense_navy }
			} 
		}	
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						naval_supply_trigger = yes
							NOT = {
								has_province_modifier = supplied_by_sea
							}
					}
				}

		#trigger_else = {
		#	custom_tooltip = {
		#		text = no_marines_tt
		#	}
		#}

	}


	start_effect = {
	
		unit_owner = {
			while = {
				count >= num_of_ships_svalue
				add_country_modifier = {
					name = naval_action
					duration = 90
					mode = add	
				}
			}
			add_country_modifier = {
				name = naval_supplier
				duration = 90
			}
		}

		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						naval_supply_trigger = yes
					}
				}
			}
			#custom_tooltip = naval_raiding_ae_gain_tooltip
			show_as_tooltip = { #All fake for the tooltips sake
				unit_location = {
						every_neighbor_province = { 
							limit = {
								naval_supply_trigger = yes
								NOT = { has_province_modifier = supplied_by_sea }
							}
							custom = naval_supply_explain_modifier_tooltip
							add_province_modifier = {
								name = supplied_by_sea
								duration = 90
							}
						}
				}
			}

			unit_location = {
				ordered_neighbor_province = {
					limit = {
						naval_supply_trigger = yes
						NOT = {	has_province_modifier = supplied_by_sea }
					}
					order_by = fort_level
					hidden_effect = {
						add_province_modifier = {
							name = supplied_by_sea
							duration = 90
						}
					}
				}
			}
		}

	}

	finished_when = {
		OR = {
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			is_moving = yes
			has_commander = no
			strength_percentage < 0.3
			morale_percentage < 0.3
		}
	}

	finish_effect = {

	}
	
	ai_will_do = {
		modifier = {
			trigger = {

				experience_percentage < 0.65
				
				unit_owner = {
					treasury > 50
					manpower > 20
				}
			}
			add = {
				value = 250
				divide = unit_size_rank
			}
		}
	}
	
	ai_will_revoke = {
		OR = {
			is_idle = no
			unit_threat > 0.2
			experience_percentage >= 0.95

			unit_owner = {
				treasury < 1
			}
			unit_owner = {
				months_to_war < 12
			}
		}
	}
}
