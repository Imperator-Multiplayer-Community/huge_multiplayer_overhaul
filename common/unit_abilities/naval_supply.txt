naval_supply = {
	duration = 91
	enable = yes
	toggle = yes
	navy_only = yes

	map = no

	soundeffect ="event:/SFX/UI/Unit/sfx_ui_unit_reorganization"

	allow = {
		is_navy = yes
		unit_size >= 25
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege = no
		has_commander = yes

		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}

		custom_tooltip = {
			text = naval_supply_explain_tt
			any_sub_unit = {
				ship_category = heavy
				subunit_strength_percentage >= 0.95
			}
		}
		NOT = {
			OR = {
				unit_owner = { has_country_modifier = naval_supplier }  #New addition, lets see if it works
				unit_owner = { has_low_economic_policy = expense_navy }
			}
		}

		trigger_if = {
			limit = {
				NOT = {
					can_naval_supply_trigger = yes
				}
			}

			custom_tooltip = explain_naval_supply_tooltip

			trigger_if = {
				limit = {
					NOT = {
						unit_location = {
							any_neighbor_province = {
								has_province_modifier = supplied_by_sea
							}
						}
					}
				}

				custom_tooltip = naval_supply_cooldown_tooltip
			}

			always = no
		}
		trigger_else = {
			can_naval_supply_trigger = yes
		}
	}


	start_effect = {
		custom_tooltip = naval_supply_explain_modifier_tooltip
		if = {
			limit = {
				can_naval_supply_trigger = yes
			}

			show_as_tooltip = { #All fake for the tooltips sake
				unit_owner = {
					add_country_modifier = {
						name = naval_supplier
						duration = 90
					}
				}

				add_unit_modifier = {
					name = naval_action
					duration = 90
				}

				unit_location = {
					ordered_neighbor_province = {
						limit = {
							naval_supply_province_trigger = yes
						}
						order_by = total_population
						max = 1

						add_province_modifier = {
							name = supplied_by_sea
							duration = 90
						}
					}
				}
			}

			hidden_effect = {
				save_scope_as = naval_supplier_unit
				unit_location = {
					save_scope_as = naval_supplier_location
				}
				set_variable = is_navy_supplying

				unit_owner = {
					add_country_modifier = {
						name = naval_supplier
						duration = 90
					}
				}

				add_unit_modifier = {
					name = naval_action
					duration = 90
				}

				trigger_event = {
					id = hmo_unit_abilities.3
					days = 90
				}
			}
		}
	}

	finished_when = {
		OR = {
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			is_moving = yes
			has_commander = no
			strength_percentage < 0.3
			morale_percentage < 0.3
			in_combat = yes
		}
	}

	finish_effect = {
		navy_supply_finish_effect = yes
	}

	ai_will_do = {
		modifier = {
			trigger = {

				experience_percentage < 0.65

				unit_owner = {
					treasury > 50
					manpower > 20
				}
			}
			add = {
				value = 250
				divide = unit_size_rank
			}
		}
	}

	ai_will_revoke = {
		OR = {
			is_idle = no
			unit_threat > 0.2
			experience_percentage >= 0.95

			unit_owner = {
				treasury < 1
			}
			unit_owner = {
				months_to_war < 12
			}
		}
	}

	idle_entity_state = combat
}
