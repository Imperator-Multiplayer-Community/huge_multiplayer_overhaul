military_colonies = {

	duration = 0
	enable = no
	toggle = no
	army_only = yes
	
	map = yes
	soundeffect ="event:/SFX/Animations/Units/Generic/sfx_anim_unit_build_colony"
	
	
	hidden = {
		always = yes
        unit_owner = { 
            NOT = { 
                is_tradition_tree_allowed = terrain_traditions 
            } 
        }
	}

	allow = {
		unit_owner = { 
			has_military_bonus = military_colonies_path_1
		}
		has_commander = yes
		is_army = yes
		in_combat = no
		is_moving = no
		in_siege = no
		is_exiled = no
		unit_size >= 5
		custom_tooltip = {
			text = "num_of_loyal_cohorts_tooltip"
			num_of_non_loyal_cohorts_svalue >= 1
		}
		hidden:unit_owner = { can_pay_price = create_military_colony }
		hidden:unit_location = {
			has_owner = yes
			NOT = { dominant_province_culture = owner.culture }
			total_population <= 5
			state_loyalty >= 33
			custom_tooltip = {
				text = military_colonies_capital_tooltip
				NOT = {
					state = {
						any_state_province = {
							is_capital = yes
						}
					}
				}
			}
			owner = root.unit_owner 
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	
	
	start_effect = {
		unit_owner = {  pay_price = create_military_colony }
		if = {
			limit = {
				exists = commander
			}
			random_sub_unit = {
				limit = {
					has_personal_loyalty = no
				}
				if = {
					limit = {
						root = {
							unit_owner = {
								is_republic = yes
							}
						}
					}
					set_personal_loyalty = root.commander
				}
				else_if = {
					limit = {
						root = {
							commander = {
								is_ruler = no
							}
						} 
					}
					set_personal_loyalty = root.commander
				}
				else = {
					root = {
						unit_owner = {
							add_tyranny = 1
						}
					}
				}
			}
		}
		unit_location = {
			add_state_loyalty = -15
			add_province_modifier = {
				name = "local_colonia"
				duration = -1
			}
		}
		unit_owner = {
			random_owned_province = {
				limit = {
					total_population >= 3
					any_pops_in_province = {
						count >= 2
						pop_culture = location.owner.culture
					}
				}
				while = {
					count = 2
					random_pops_in_province = {
						limit = {
							pop_culture = location.owner.culture
						}
						move_pop = root.unit_location
					}
				}
			}
		}
	}
}
