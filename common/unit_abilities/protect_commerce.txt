protect_commerce = {

	duration = 90
	enable = yes
	toggle = no
	navy_only = yes
	cancel_on_combat_end = yes

	map = yes

	soundeffect ="event:/SFX/UI/Unit/sfx_ui_unit_reorganization"

	allow = {
		
		is_navy = yes
		unit_size >= 10
		in_combat = no
		is_moving = no
		is_exiled = no
		in_siege = no
		has_commander = yes
		# custom_tooltip = {
		# 	text = marines_ratio
		# 	num_of_marines >= num_of_ships_10
		# }

		trigger_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							trading_partner_province_trigger = yes
						}
					}
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						owner = {
							OR = {
								exports_to = root.unit_owner
								imports_from = root.unit_owner
							}
						}
						NOT = {
							OR = {
				 				has_province_modifier = mugged_by_reality 
								has_province_modifier = trading_partner
				 			} 
						}
						owner = controller
							NOT = {
								owner = root.unit_owner 
							}
					}
				}
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							trading_partner_province_trigger = yes
						}
					}
				}
			}
			custom_tooltip = {
				text = explain_trading_tooltip
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				unit_location = {
					NOT = {
						any_neighbor_province = {
							is_sea = no
							trading_partner_province_trigger = yes
							NOT = {
								has_province_modifier = trading_partner
							}
						}
					}
				}
			}
			custom_tooltip = {
				text = trading_cooldown_tooltip
				always = no
			}
		}

		NOT = {
			OR = {
				unit_owner = { has_country_modifier = protect_merchants }  #New addition, lets see if it works
				unit_owner = { has_low_economic_policy = expense_navy }
			} 
		}		
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
		unit_location = {
			any_neighbor_province = {
				is_sea = no
				trading_partner_province_trigger = yes
				NOT = {
					has_province_modifier = trading_partner
				}
			}
		}
		#trigger_else = {
		#	custom_tooltip = {
		#		text = no_marines_tt
		#	}
		#}
	}
	

	start_effect = {

		unit_owner = {
			while = {
				count >= num_of_ships_svalue  #num_of_ships_svalue #This script value is not working. 
							#A number will work, but it will not adjust to the ships in the fleet
				add_country_modifier = {
					name = protect_merchants
					duration = 90
					mode = add
				}
			}
		}

		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						trading_partner_province_trigger = yes
					}
				}
			}
			#custom_tooltip = trading_ae_gain_tooltip
			show_as_tooltip = { 
				unit_location = {
					every_neighbor_province = {
						limit = {
							trading_partner_province_trigger = yes
							NOT = {	has_province_modifier = trading_partner }
						}
						custom = trading_explain_modifier_tooltip
						add_province_modifier = {
							name = trading_partner
							duration = 180
						}
					}
				}
			}
	
			unit_location = {
				ordered_neighbor_province = {
					limit = {
						trading_partner_province_trigger = yes
						NOT = {	has_province_modifier = trading_partner }
					}
					order_by = total_population
					hidden_effect = {
						add_province_modifier = {
							name = trading_partner
							duration = 180
						}
					}
				}
			}
		}

	}

	finished_when = {
		OR = {
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			is_moving = yes
			has_commander = no
			strength_percentage < 0.3
			morale_percentage < 0.3
		}
	}

	finish_effect = {

	}
	
	ai_will_do = {
		modifier = {
			trigger = {

				experience_percentage < 0.65
				
				unit_owner = {
					treasury > 50
					manpower > 20
				}
			}
			add = {
				value = 250
				divide = unit_size_rank
			}
		}
	}
	
	ai_will_revoke = {
		OR = {
			is_idle = no
			unit_threat > 0.2
			experience_percentage >= 0.95

			unit_owner = {
				treasury < 1
			}
			unit_owner = {
				months_to_war < 12
			}
		}
	}
}
