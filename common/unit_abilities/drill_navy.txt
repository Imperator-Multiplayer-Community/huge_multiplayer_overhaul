drill_navy = {
	duration = -1
	enable = no
	toggle = yes
	navy_only = yes
	cancel_on_combat_end = yes
	
	map = yes
	
	soundeffect ="event:/SFX/UI/Unit/sfx_ui_unit_reorganization"


	allow = {
		is_navy = yes
		in_combat = no
		is_moving = no
		has_commander = yes		
		NOT = { unit_owner = { has_low_economic_policy = expense_navy } }

		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
	}
	modifier = {
		monthly_experience_gain = 2.5
		loyalty_gain_chance = 1.0
		navy_maintenance_cost = 0.33
	}
	
	
	start_effect = {
		set_variable = is_navy_drilling
	}
	
	finish_effect = {
		if = {
			limit = {
				has_variable = is_navy_drilling
			}

			remove_variable = is_navy_drilling
		}
	}
	
	finished_when = {
		OR = {
			in_combat = yes
			is_moving = yes
			is_exiled = yes
			unit_owner = { has_low_economic_policy = expense_navy }
			has_commander = no
			navy_fishing = yes
			navy_slave_raiding = yes
			navy_supplying = yes
			navy_assaulting = yes
			navy_raiding = yes
			navy_raiding_commerce = yes
			navy_protecting_commerce = yes
		}
	}
	
	ai_will_do = {
		modifier = {
			trigger = {

				experience_percentage < 0.65
				
				unit_owner = {
					treasury > 50
					manpower > 20
				}
			}
			add = {
				value = 250
				divide = unit_size_rank
			}
		}
	}
	
	ai_will_revoke = {
		OR = {
			is_idle = no
			unit_threat > 0.2
			experience_percentage >= 0.95

			unit_owner = {
				treasury < 1
			}
			unit_owner = {
				months_to_war < 12
			}
		}
	}
}
