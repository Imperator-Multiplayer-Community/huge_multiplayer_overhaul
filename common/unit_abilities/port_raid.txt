port_raid = {

	enable = yes
	toggle = no
	soundeffect = "event:/SFX/UI/Unit/sfx_ui_unit_port_raid"

	map = yes

	navy_only = yes

	allow = {
		is_navy = yes
		in_combat = no
		is_moving = no
		is_exiled = no
		custom_tooltip = {
			text = port_raid_explain_tt
			any_sub_unit = {
				ship_category = heavy
				subunit_strength_percentage >= 0.95
			}
			# text = marines_ratio
			# num_of_marines >= num_of_ships_10
		}
		trigger_if = {
			limit = {
				exists = commander
			}
			commander = {
				loyalty > 33
			}
		}
		trigger_if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						has_siege = yes
						siege = {
							has_breach = no
						}
					}
				}
			}
			unit_location = {
				any_neighbor_province = {
					is_sea = no
					has_owner = yes
					is_port = yes
					NOT = { controller = root.unit_owner }
					controller = { war_with = root.unit_owner }
					has_siege = yes
					siege = {
						has_breach = no
					}
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = port_raid_no_valid_port
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						has_siege = yes
						siege = {
							has_breach = no
						}
					}
				}	
			}

			#custom_tooltip = {
			#	text = no_marines_tt
			#}
		}
	}
	
	
	
	start_effect = {
		random_sub_unit = {
			limit = {
				ship_category = heavy
				subunit_strength_percentage >= 0.95
			}
			add_subunit_strength = -0.30
		}
		if = {
			limit = {
				unit_location = {
					any_neighbor_province = {
						is_sea = no
						has_owner = yes
						is_port = yes
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						has_siege = yes
						siege = {
							has_breach = no
						}
					}
				}
			}
			hidden_effect = {
				root.unit_owner = {

					add_country_modifier = {
						name = naval_action
						duration = 45
					}
				}
			}
			unit_location = {
				random_neighbor_province = {
					limit = {
						is_sea = no
						has_owner = yes
						is_port = yes
						NOT = { controller = root.unit_owner }
						controller = { war_with = root.unit_owner }
						has_siege = yes
						siege = {
							has_breach = no
						}
					}
					siege = { add_breach = 1 }
				}
			}
		}
		else = {
			custom_tooltip = port_raid_explain_effect_tt
		}
	}
	
	
}
