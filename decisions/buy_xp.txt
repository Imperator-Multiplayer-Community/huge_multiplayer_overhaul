country_decisions = {
	
	# Borrow some XP to hire mercs
	buy_xp = {
	
		potential = {
			num_of_cities >= 1
			NOT = {
				has_variable = bought_xp
			}
		}
		
		allow = {
			political_influence >= 15
			OR = {
				is_tribal = yes
				is_monarchy = yes
				is_republic = yes
			}
		}
		
		effect = {
			add_political_influence = -15
			add_military_experience = 5
			set_variable = {
				name = bought_xp
				days = -1
			}
		}
		
		ai_will_do = {
			base = 100
		}
	}

}
