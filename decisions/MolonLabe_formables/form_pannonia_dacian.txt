			}

		}
		
		highlight = {

				OR = {
					is_in_region = pannonia_region
					is_in_region = dacia_region
				}
			}
		}
		
		allow = {
			custom_tooltip = {
				text = pannonia_not_exists

			}

				change_country_flag = PANNONIA_DACIAN_FLAG

					id = formable_heritage.130
				}
			}
			add_5_free_province_investments = yes
		}
		
		ai_will_do = {
			base = 1
			modifier = {
				add = 10

			}

	}
